def timer(name)
  metric_name = new_metric_name(sanitize_classname(self.class.name), name, "timer")
  new_timer = metrics_registry.newTimer(metric_name,
                                        DEFAULT_DURATION_UNIT,
                                        DEFAULT_RATE_UNIT)
  captured_timer = new_timer.time
  begin
    yield new_timer
  ensure
    captured_timer.stop
  end
end
